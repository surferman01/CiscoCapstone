# CiscoCapstone
Capstone Project for client James Turman @ Cisco

# Getting Started

cd app

python3 -m venv .venv
or
python -m venv .venv

source .venv/bin/activate
or
.venv\Scripts\activate

pip install -r requirements.txt

python3 main.py
or
python main.py

# Build single executable

./build_executable.sh

The packaged app will be created at:

dist/CSFC
or on Windows:
dist\CSFC.exe
